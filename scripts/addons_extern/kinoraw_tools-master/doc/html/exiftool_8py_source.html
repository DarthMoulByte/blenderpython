22 # PyExifTool is a Python library to communicate with an instance of Phil
23 # Harvey's excellent ExifTool_ command-line application. The library
24 # provides the class :py:class:`ExifTool` that runs the command-line
25 # tool in batch mode and features methods to send commands to that
26 # program, including methods to extract meta-information from one or
27 # more image files. Since ``exiftool`` is run in batch mode, only a
28 # single instance needs to be launched and can be reused for many
29 # queries. This is much more efficient than launching a separate
30 # process for every single query.
31 #
32 # .. _ExifTool: http://www.sno.phy.queensu.ca/~phil/exiftool/
33 #
34 # The source code can be checked out from the github repository with
35 #
36 # ::
37 #

40 # Alternatively, you can download a tarball_. There haven't been any
41 # releases yet.

93 def _fscodec():
94  encoding = sys.getfilesystemencoding()
95  errors = "strict"
96  if encoding != "mbcs":
97  try:
98  codecs.lookup_error("surrogateescape")
99  except LookupError:
100  pass
101  else:
102  errors = "surrogateescape"

104  ##
105  #
106  # Encode filename to the filesystem encoding with 'surrogateescape' error
107  # handler, return bytes unchanged. On Windows, use 'strict' error handler
108  # if the file system encoding is 'mbcs' (which is the default encoding).
109  #
110  def fsencode(filename):
111  if isinstance(filename, bytes):
112  return filename
113  else:
114  return filename.encode(encoding, errors)
115 
116  return fsencode

123 # Run the `exiftool` command-line tool and communicate to it.
124 #
125 # You can pass the file name of the ``exiftool`` executable as an
126 # argument to the constructor. The default value ``exiftool`` will
127 # only work if the executable is in your ``PATH``.
128 #
129 # Most methods of this class are only available after calling
130 # :py:meth:`start()`, which will actually launch the subprocess. To
131 # avoid leaving the subprocess running, make sure to call
132 # :py:meth:`terminate()` method when finished using the instance.
133 # This method will also be implicitly called when the instance is
134 # garbage collected, but there are circumstance when this won't ever
135 # happen, so you should not rely on the implicit process
136 # termination. Subprocesses won't be automatically terminated if
137 # the parent process exits, so a leaked subprocess will stay around
138 # until manually killed.
139 #
140 # A convenient way to make sure that the subprocess is terminated is
141 # to use the :py:class:`ExifTool` instance as a context manager::
142 #
143 # with ExifTool() as et:
144 # ...
145 #
146 # .. warning:: Note that there is no error handling. Nonsensical
147 # options will be silently ignored by exiftool, so there's not
148 # much that can be done in that regard. You should avoid passing
149 # non-existent files to any of the methods, since this will lead
150 # to undefied behaviour.
151 #
152 # .. py:attribute:: running
153 #
154 # A Boolean value indicating whether this instance is currently
155 # associated with a running subprocess.
156 #
157 class ExifTool(object):
158 
159  def __init__(self, executable_=None):
160  if executable_ is None:
161  self.executable = executable
162  else:
163  self.executable = executable_
164  self.running = False
165 
166  ##
167  # Start an ``exiftool`` process in batch mode for this instance.
168  #
169  # This method will issue a ``UserWarning`` if the subprocess is
170  # already running. The process is started with the ``-G`` and
171  # ``-n`` as common arguments, which are automatically included
172  # in every command you run with :py:meth:`execute()`.
173  #
174  def start(self):
175  if self.running:
176  warnings.warn("ExifTool already running; doing nothing.")
177  return
178  with open(os.devnull, "w") as devnull:
179  self._process = subprocess.Popen(
180  [self.executable, "-stay_open", "True", "-@", "-",
181  "-common_args", "-G", "-u", "-a", "-n"],
182  stdin=subprocess.PIPE, stdout=subprocess.PIPE,
183  stderr=devnull)
184  self.running = True
185 
186  ##
187  # Terminate the ``exiftool`` process of this instance.
188  #
189  # If the subprocess isn't running, this method will do nothing.
190  #
191  def terminate(self):
192  if not self.running:
193  return
194  self._process.stdin.write(b"-stay_open\nFalse\n")
195  self._process.stdin.flush()
196  self._process.communicate()
197  del self._process
198  self.running = False
199 
200  def __enter__(self):
201  self.start()
202  return self
203 
204  def __exit__(self, exc_type, exc_val, exc_tb):
205  self.terminate()
206 
207  def __del__(self):
208  self.terminate()

210  ##
211  # Execute the given batch of parameters with ``exiftool``.
212  #
213  # This method accepts any number of parameters and sends them to
214  # the attached ``exiftool`` process. The process must be
215  # running, otherwise ``ValueError`` is raised. The final
216  # ``-execute`` necessary to actually run the batch is appended
217  # automatically; see the documentation of :py:meth:`start()` for
218  # the common options. The ``exiftool`` output is read up to the
219  # end-of-output sentinel and returned as a raw ``bytes`` object,
220  # excluding the sentinel.
221  #
222  # The parameters must also be raw ``bytes``, in whatever
223  # encoding exiftool accepts. For filenames, this should be the
224  # system's filesystem encoding.
225  #
226  # .. note:: This is considered a low-level method, and should
227  # rarely be needed by application developers.
228  #
229  def execute(self, *params):
230  if not self.running:
231  raise ValueError("ExifTool instance not running.")
232  self._process.stdin.write(b"\n".join(params + (b"-execute\n",)))
233  self._process.stdin.flush()
234  output = b""
235  fd = self._process.stdout.fileno()
236  while not output[-32:].strip().endswith(sentinel):
237  output += os.read(fd, block_size)
238  return output.strip()[:-len(sentinel)]

240  ##
241  # Execute the given batch of parameters and parse the JSON output.
242  #
243  # This method is similar to :py:meth:`execute()`. It
244  # automatically adds the parameter ``-j`` to request JSON output
245  # from ``exiftool`` and parses the output. The return value is
246  # a list of dictionaries, mapping tag names to the corresponding
247  # values. All keys are Unicode strings with the tag names
248  # including the ExifTool group name in the format <group>:<tag>.
249  # The values can have multiple types. All strings occurring as
250  # values will be Unicode strings. Each dictionary contains the
251  # name of the file it corresponds to in the key ``"SourceFile"``.
252  #
253  # The parameters to this function must be either raw strings
254  # (type ``str`` in Python 2.x, type ``bytes`` in Python 3.x) or
255  # Unicode strings (type ``unicode`` in Python 2.x, type ``str``
256  # in Python 3.x). Unicode strings will be encoded using
257  # system's filesystem encoding. This behaviour means you can
258  # pass in filenames according to the convention of the
259  # respective Python version – as raw strings in Python 2.x and
260  # as Unicode strings in Python 3.x.
261  #
262  def execute_json(self, *params):
263  params = map(fsencode, params)
264  return json.loads(self.execute(b"-j", *params).decode("utf-8"))

285  # Return only specified tags for the given files.
286  #
287  # The first argument is an iterable of tags. The tag names may
288  # include group names, as usual in the format <group>:<tag>.
289  #
290  # The second argument is an iterable of file names.
291  #
292  # The format of the return value is the same as for
293  # :py:meth:`execute_json()`.
294  #
295  def get_tags_batch(self, tags, filenames):
296  # Explicitly ruling out strings here because passing in a
297  # string would lead to strange and hard-to-find errors
298  if isinstance(tags, basestring):
299  raise TypeError("The argument 'tags' must be "
300  "an iterable of strings")
301  if isinstance(filenames, basestring):
302  raise TypeError("The argument 'filenames' must be "
303  "an iterable of strings")
304  params = ["-" + t for t in tags]
305  params.extend(filenames)
306  return self.execute_json(*params)

317  ##
318  # Extract a single tag from the given files.
319  #
320  # The first argument is a single tag name, as usual in the
321  # format <group>:<tag>.
322  #
323  # The second argument is an iterable of file names.
324  #
325  # The return value is a list of tag values or ``None`` for
326  # non-existent tags, in the same order as ``filenames``.
327  #
328  def get_tag_batch(self, tag, filenames):
329  data = self.get_tags_batch([tag], filenames)
330  result = []
331  for d in data:
332  d.pop("SourceFile")
333  result.append(next(iter(d.values()), None))
334  return result

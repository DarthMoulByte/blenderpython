321  remove = False

343 def randompartition(lst,n,rand):
344  division = len(lst) / float(n)
345  lista = []
346  for i in range(n): lista.append(division)
347  var=0
348  for i in range(n-1):
349  lista[i]+= random.randint(-int(rand*division),int(rand*division))
350  var+=lista[i]
351  if lista[n-1] != len(lst)-var:
352  lista[n-1] = len(lst)-var
353  random.shuffle(lista)
354  division = len(lst) / float(n)
355  count = 0
356  newlist=[]
357  for i in range(n):
358  #print(lst[count : int(lista[i]-1)+count])
359  newlist.append([lst[count : int(lista[i]-1)+count]])
360  count += int(lista[i])
361  return newlist

157  _frame_rate_args_prev = None

160  @staticmethod
161  def _draw_framerate_label(*args):
162  # avoids re-creating text string each draw
163  if JumptoCut._frame_rate_args_prev == args:
164  return JumptoCut._frame_rate_ret
165 
166  fps, fps_base, preset_label = args
167 
168  if fps_base == 1.0:
169  fps_rate = round(fps)
170  else:
171  fps_rate = round(fps / fps_base, 2)
172 
173  # TODO: Change the following to iterate over existing presets
174  custom_framerate = (fps_rate not in {23.98, 24, 25, 29.97, 30, 50, 59.94, 60})
175 
176  if custom_framerate is True:
177  fps_label_text = "Custom (%r fps)" % fps_rate
178  show_framerate = True
179  else:
180  fps_label_text = "%r fps" % fps_rate
181  show_framerate = (preset_label == "Custom")
182 
183  JumptoCut._frame_rate_args_prev = args
184  JumptoCut._frame_rate_ret = args = (fps_label_text, show_framerate)
185  return args
186

192  args = rd.fps, rd.fps_base, JumptoCut._preset_class.bl_label

195  sub.menu("RENDER_MT_framerate_presets", text=fps_label_text)
